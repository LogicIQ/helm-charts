1. Check the status of the secret-santa controller:
   kubectl get pods -n {{ .Release.Namespace }} -l app.kubernetes.io/name={{ include "secret-santa.name" . }}

2. View the logs:
   kubectl logs -n {{ .Release.Namespace }} -l app.kubernetes.io/name={{ include "secret-santa.name" . }}

3. Create a SecretSanta resource to generate secrets:
   kubectl apply -f - <<EOF
   apiVersion: secrets.secret-santa.io/v1alpha1
   kind: SecretSanta
   metadata:
     name: example-secret
     namespace: {{ .Release.Namespace }}
   spec:
     secretName: my-generated-secret
     generators:
     - name: password
       type: random_password
       config:
         length: 16
     template: |
       password: {{ "{{" }} .password {{ "}}" }}
   EOF

4. Check the generated secret:
   kubectl get secret my-generated-secret -n {{ .Release.Namespace }} -o yaml
